void addUnit(Unit unit) {
  if (unit.origin.card.arena != arena) {
    throw ArgumentError.value(
      unit,
      'unit',
      ''
          'Unit arena (${unit.origin.card.arena.name}) does not match shared '
          'zone arena (${arena.name}).',
    );
  }
  _units.add(unit);
}
